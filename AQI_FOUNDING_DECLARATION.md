def compute_delta(previous: TalkPacket, current: TalkPacket) -> DeltaPacket:

    # Payload comparison

        changes.append(Change(
            change_type=ChangeType.MODIFY,
            field="payload",
            old_value=previous.state_change.payload,
            new_value=current.state_change.payload,

        id=f"delta_{uuid.uuid4().hex[:8]}",

def atomic_write(filepath: str):
    """Context manager for atomic file operations."""
    temp_path = f"{filepath}.tmp"
    try:
        with open(temp_path, 'w') as f:
            yield f
        os.rename(temp_path, filepath)  # Atomic move
    except Exception:

## **4.5 Governance Intelligence Capabilities**

### **Integrity Verification**

Cryptographic hash chaining:

```python
def compute_hash_chain(run_id: str) -> List[str]:
    """Generate hash chain for integrity verification."""
    packet_files = sorted(glob.glob(f"aqi_storage/runs/{run_id}/packets/*.json"))

    hashes = []
    for packet_file in packet_files:
        with open(packet_file, 'r') as f:
            packet_data = json.load(f)
            packet_hash = hashlib.sha256(
                json.dumps(packet_data, sort_keys=True).encode()
            ).hexdigest()
            hashes.append(packet_hash)

    return hashes

def compute_merkle_root(run_id: str) -> str:
    """Build Merkle tree for tamper detection."""
    hashes = compute_hash_chain(run_id)

    if not hashes:
        return None

    # Build Merkle tree
    layer = hashes
    while len(layer) > 1:
        next_layer = []
        for i in range(0, len(layer), 2):
            left = layer[i]
            right = layer[i + 1] if i + 1 < len(layer) else layer[i]
            combined = hashlib.sha256((left + right).encode()).hexdigest()
            next_layer.append(combined)
        layer = next_layer

    return layer[0]
```
